176 /* Sets up system pin muxing */
177 void Board_SetupMuxing(void)
178 {
179  int i, j;
180 
181  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_EMC);
182  Chip_SYSCTL_PeriphReset(SYSCTL_RESET_IOCON);
183 
184  /* Setup data, address, and EMC control pins with high slew rate */
185  for (i = 3; i <= 4; i++) {
186  for (j = 0; j <= 31; j++) {
187  Chip_IOCON_PinMuxSet(LPC_IOCON, (uint8_t) i, (uint8_t) j, (IOCON_FUNC1 | IOCON_FASTSLEW_EN));
188  }
189  }
190  for (i = 16; i <= 31; i++) {
191  Chip_IOCON_PinMuxSet(LPC_IOCON, 2, (uint8_t) i, (IOCON_FUNC1 | IOCON_FASTSLEW_EN));
192  }
193 
194  /* Setup system level pin muxing */
195  Chip_IOCON_SetPinMuxing(LPC_IOCON, pinmuxing, sizeof(pinmuxing) / sizeof(PINMUX_GRP_T));
196 }

198 /* Setup system clocking */
199 void Board_SetupClocking(void)
200 {
201  /* CPU clock source starts with IRC */
202  /* Enable PBOOST for CPU clock over 100MHz */
203  Chip_SYSCTL_EnableBoost();
204 
205  Chip_SetupXtalClocking();
206 
207  /* SPIFI clocking will be derived from Main PLL with a divider of 2 (60MHz) */
208  Chip_Clock_SetSPIFIClockDiv(2);
209  Chip_Clock_SetSPIFIClockSource(SYSCTL_SPIFICLKSRC_MAINPLL);
210 }

212 /* Setup external memories */
213 void Board_SetupExtMemory(void)
214 {
215  /* Setup EMC Delays */
216  /* Move all clock delays together */
217  LPC_SYSCTL->EMCDLYCTL = (CLK0_DELAY) | (CLK0_DELAY << 8) | (CLK0_DELAY << 16 | (CLK0_DELAY << 24));
218 
219  /* Setup EMC Clock Divider for divide by 2 */
220  /* Setup EMC clock for a divider of 2 from CPU clock. Enable EMC clock for
221  external memory setup of DRAM. */
222  Chip_Clock_SetEMCClockDiv(SYSCTL_EMC_DIV2);
223  Chip_SYSCTL_PeriphReset(SYSCTL_RESET_EMC);
224 
225  /* Init EMC Controller -Enable-LE mode- clock ratio 1:1 */
226  Chip_EMC_Init(1, 0, 0);
227 
228  /* Init EMC Dynamic Controller */
229  Chip_EMC_Dynamic_Init((IP_EMC_DYN_CONFIG_T *) &IS42S32800D_config);
230 
231  /* Init EMC Static Controller CS0 */
232  Chip_EMC_Static_Init((IP_EMC_STATIC_CONFIG_T *) &SST39VF320_config);
233 
234 
235  /* Init EMC Static Controller CS1 */
236  Chip_EMC_Static_Init((IP_EMC_STATIC_CONFIG_T *) &K9F1G_config);
237 
238  /* EMC Shift Control */
239  LPC_SYSCTL->SCS |= 1;
240 }

243 void Board_SystemInit(void)
