205 /* Transmit a byte array through the UART peripheral (non-blocking) */
206 int Chip_UART_Send(LPC_USART_T *pUART, const void *data, int numBytes)
207 {
208  int sent = 0;
209  uint8_t *p8 = (uint8_t *) data;
210 
211  /* Send until the transmit FIFO is full or out of bytes */
212  while ((sent < numBytes) &&
213  ((Chip_UART_ReadLineStatus(pUART) & UART_LSR_THRE) != 0)) {
214  Chip_UART_SendByte(pUART, *p8);
215  p8++;
216  sent++;
217  }
218 
219  return sent;
220 }

282 /* Determines and sets best dividers to get a target bit rate */
283 uint32_t Chip_UART_SetBaud(LPC_USART_T *pUART, uint32_t baudrate)
284 {
285  uint32_t div, divh, divl, clkin;
286 
287  /* Determine UART clock in rate without FDR */
288 #if defined(CHIP_LPC175X_6X)
289  clkin = Chip_Clock_GetPeripheralClockRate(Chip_UART_GetClkIndex(pUART));
290 #else
291  clkin = Chip_Clock_GetPeripheralClockRate();
292 #endif
293  div = clkin / (baudrate * 16);
294 
295  /* High and low halves of the divider */
296  divh = div / 256;
297  divl = div - (divh * 256);
298 
299  Chip_UART_EnableDivisorAccess(pUART);
300  Chip_UART_SetDivisorLatches(pUART, divl, divh);
301  Chip_UART_DisableDivisorAccess(pUART);
302 
303  /* Fractional FDR already setup for 1 in UART init */
304 
305  return clkin / div;
306 }

330 /* Populate a transmit ring buffer and start UART transmit */
331 uint32_t Chip_UART_SendRB(LPC_USART_T *pUART, RINGBUFF_T *pRB, const void *data, int bytes)
332 {
333  uint32_t ret;
334  uint8_t *p8 = (uint8_t *) data;
335 
336  /* Don't let UART transmit ring buffer change in the UART IRQ handler */
337  Chip_UART_IntDisable(pUART, UART_IER_THREINT);
338 
339  /* Move as much data as possible into transmit ring buffer */
340  ret = RingBuffer_InsertMult(pRB, p8, bytes);
341  Chip_UART_TXIntHandlerRB(pUART, pRB);
342 
343  /* Add additional data to transmit ring buffer if possible */
344  ret += RingBuffer_InsertMult(pRB, (p8 + ret), (bytes - ret));
345 
346  /* Enable UART transmit interrupt */
347  Chip_UART_IntEnable(pUART, UART_IER_THREINT);
348 
349  return ret;
350 }

380 /* Determines and sets best dividers to get a target baud rate */
381 uint32_t Chip_UART_SetBaudFDR(LPC_USART_T *pUART, uint32_t baudrate)
382 
383 {
384  uint32_t uClk;
385  uint32_t actualRate = 0, d, m, bestd, bestm, tmp;
386  uint32_t current_error, best_error;
387  uint64_t best_divisor, divisor;
388  uint32_t recalcbaud;
389 
390  /* Get Clock rate */
391 #if defined(CHIP_LPC175X_6X)
392  uClk = Chip_Clock_GetPeripheralClockRate(Chip_UART_GetClkIndex(pUART));
393 #else
394  uClk = Chip_Clock_GetPeripheralClockRate();
395 #endif
396 
397  /* In the Uart IP block, baud rate is calculated using FDR and DLL-DLM registers
398  * The formula is :
399  * BaudRate= uClk * (mulFracDiv/(mulFracDiv+dividerAddFracDiv) / (16 * (DLL)
400  * It involves floating point calculations. That's the reason the formulae are adjusted with
401  * Multiply and divide method.*/
402  /* The value of mulFracDiv and dividerAddFracDiv should comply to the following expressions:
403  * 0 < mulFracDiv <= 15, 0 <= dividerAddFracDiv <= 15 */
404  best_error = 0xFFFFFFFF;/* Worst case */
405  bestd = 0;
406  bestm = 0;
407  best_divisor = 0;
408  for (m = 1; m <= 15; m++) {
409  for (d = 0; d < m; d++) {
410 
411  /* The result here is a fixed point number. The integer portion is in the upper 32 bits.
412  * The fractional portion is in the lower 32 bits.
413  */
414  divisor = ((uint64_t) uClk << 28) * m / (baudrate * (m + d));
415 
416  /* The fractional portion is the error. */
417  current_error = divisor & 0xFFFFFFFF;
418 
419  /* Snag the integer portion of the divisor. */
420  tmp = divisor >> 32;
421 
422  /* If closer to the next divisor... */
423  if (current_error > ((uint32_t) 1 << 31)) {
424 
425  /* Increment to the next divisor... */
426  tmp++;
427 
428  /* Now the error is the distance to the next divisor... */
429  current_error = -current_error;
430  }
431 
432  /* Can't use a divisor that's less than 1 or more than 65535. */
433  if ((tmp < 1) || (tmp > 65535)) {
434  /* Out of range */
435  continue;
436  }
437 
438  /* Also, if fractional divider is enabled can't use a divisor that is less than 3. */
439  if ((d != 0) && (tmp < 3)) {
440  /* Out of range */
441  continue;
442  }
443 
444  /* Do we have a new best? */
445  if (current_error < best_error) {
446  best_error = current_error;
447  best_divisor = tmp;
448  bestd = d;
449  bestm = m;
450 
451  /* If error is 0, that's perfect. We're done. */
452  if (best_error == 0) {
453  break;
454  }
455  }
456  } /* for (d) */
457 
458  /* If error is 0, that's perfect. We're done. */
459  if (best_error == 0) {
460  break;
461  }
462  } /* for (m) */
463 
464  if (best_divisor == 0) {
465  /* can not find best match */
466  return 0;
467  }
468 
469  recalcbaud = (uClk >> 4) * bestm / (best_divisor * (bestm + bestd));
470 
471  /* reuse best_error to evaluate baud error */
472  if (baudrate > recalcbaud) {
473  best_error = baudrate - recalcbaud;
474  }
475  else {
476  best_error = recalcbaud - baudrate;
477  }
478 
479  best_error = (best_error * 100) / baudrate;
480 
481  /* Update UART registers */
482  Chip_UART_EnableDivisorAccess(pUART);
483  Chip_UART_SetDivisorLatches(pUART, UART_LOAD_DLL(best_divisor), UART_LOAD_DLM(best_divisor));
484  Chip_UART_DisableDivisorAccess(pUART);
485 
486  /* Set best fractional divider */
487  pUART->FDR = (UART_FDR_MULVAL(bestm) | UART_FDR_DIVADDVAL(bestd));
488 
489  /* Return actual baud rate */
490  actualRate = recalcbaud;
491 
492  return actualRate;
493 }
