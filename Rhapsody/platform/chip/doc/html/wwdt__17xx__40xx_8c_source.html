62 /* Clear WWDT interrupt status flags */
63 void Chip_WWDT_ClearStatusFlag(LPC_WWDT_T *pWWDT, uint32_t status)
64 {
65  if (status & WWDT_WDMOD_WDTOF) {
66  pWWDT->MOD &= (~WWDT_WDMOD_WDTOF) & WWDT_WDMOD_BITMASK;
67  }
68 
69  if (status & WWDT_WDMOD_WDINT) {
70  pWWDT->MOD |= WWDT_WDMOD_WDINT;
71  }
