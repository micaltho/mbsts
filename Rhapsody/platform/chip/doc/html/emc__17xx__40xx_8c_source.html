62 /* Calculate Clock Count from Timing Unit(nanoseconds) */
63 STATIC uint32_t convertTimmingParam(uint32_t EMC_Clock, int32_t input_ns, uint32_t adjust)
64 {
65  uint32_t temp;
66  if (input_ns < 0) {
67  return (-input_ns) >> 8;
68  }
69  temp = EMC_Clock / 1000000; /* MHz calculation */
70  temp = temp * input_ns / 1000;
71 
72  /* round up */
73  temp += 0xFF;
74 
75  /* convert to simple integer number format */
76  temp >>= 8;
77  if (temp > adjust) {
78  return temp - adjust;
79  }
80 
81  return 0;
82 }

84 /* Get Dynamic Memory Device Colum len */
85 STATIC uint32_t getColsLen(uint32_t DynConfig)
86 {
87  uint32_t DevBusWidth;
88  DevBusWidth = (DynConfig >> EMC_DYN_CONFIG_DEV_BUS_BIT) & 0x03;
89  if (DevBusWidth == 2) {
90  return 8;
91  }
92  else if (DevBusWidth == 1) {
93  return ((DynConfig >> (EMC_DYN_CONFIG_DEV_SIZE_BIT + 1)) & 0x03) + 8;
94  }
95  else if (DevBusWidth == 0) {
96  return ((DynConfig >> (EMC_DYN_CONFIG_DEV_SIZE_BIT + 1)) & 0x03) + 9;
97  }
98 
99  return 0;
100 }

102 /* Initializes the Dynamic Controller according to the specified parameters
103  in the IP_EMC_DYN_CONFIG_T */
104 void initDynMem(LPC_EMC_T *pEMC, IP_EMC_DYN_CONFIG_T *Dynamic_Config, uint32_t EMC_Clock)
105 {
106  uint32_t ChipSelect, tmpclk;
107  int i;
108 
109  for (ChipSelect = 0; ChipSelect < 4; ChipSelect++) {
110  LPC_EMC_T *EMC_Reg_add = (LPC_EMC_T *) ((uint32_t) pEMC + (ChipSelect << 5));
111 
112  EMC_Reg_add->DYNAMICRASCAS0 = Dynamic_Config->DevConfig[ChipSelect].RAS |
113  ((Dynamic_Config->DevConfig[ChipSelect].ModeRegister <<
114  (8 - EMC_DYN_MODE_CAS_BIT)) & 0xF00);
115  EMC_Reg_add->DYNAMICCONFIG0 = Dynamic_Config->DevConfig[ChipSelect].DynConfig;
116  }
117  pEMC->DYNAMICREADCONFIG = Dynamic_Config->ReadConfig; /* Read strategy */
118 
119  pEMC->DYNAMICRP = convertTimmingParam(EMC_Clock, Dynamic_Config->tRP, 1);
120  pEMC->DYNAMICRAS = convertTimmingParam(EMC_Clock, Dynamic_Config->tRAS, 1);
121  pEMC->DYNAMICSREX = convertTimmingParam(EMC_Clock, Dynamic_Config->tSREX, 1);
122  pEMC->DYNAMICAPR = convertTimmingParam(EMC_Clock, Dynamic_Config->tAPR, 1);
123  pEMC->DYNAMICDAL = convertTimmingParam(EMC_Clock, Dynamic_Config->tDAL, 0);
124  pEMC->DYNAMICWR = convertTimmingParam(EMC_Clock, Dynamic_Config->tWR, 1);
125  pEMC->DYNAMICRC = convertTimmingParam(EMC_Clock, Dynamic_Config->tRC, 1);
126  pEMC->DYNAMICRFC = convertTimmingParam(EMC_Clock, Dynamic_Config->tRFC, 1);
127  pEMC->DYNAMICXSR = convertTimmingParam(EMC_Clock, Dynamic_Config->tXSR, 1);
128  pEMC->DYNAMICRRD = convertTimmingParam(EMC_Clock, Dynamic_Config->tRRD, 1);
129  pEMC->DYNAMICMRD = convertTimmingParam(EMC_Clock, Dynamic_Config->tMRD, 1);
130 
131  /* TIM_Waitus(100); */
132  /*FIXME: if Timer driver is ready, it should replace below "for" delay technic */
133  for (i = 0; i < 1000; i++) { /* wait 100us */
134  }
135  pEMC->DYNAMICCONTROL = 0x00000183; /* Issue NOP command */
136 
137  /* TIM_Waitus(200); */ /* wait 200us */
138  /*FIXME: if Timer driver is ready, it should replace below "for" delay technic */
139  for (i = 0; i < 1000; i++) {}
140  pEMC->DYNAMICCONTROL = 0x00000103; /* Issue PALL command */
141 
142  pEMC->DYNAMICREFRESH = 2; /* ( 2 * 16 ) -> 32 clock cycles */
143 
144  /* FIXME: TIM_Waitus(200); */ /* wait 200us */
145  for (i = 0; i < 80; i++) {}
146 
147  tmpclk = EMC_DIV_ROUND_UP(convertTimmingParam(EMC_Clock, Dynamic_Config->RefreshPeriod, 0), 16);
148  pEMC->DYNAMICREFRESH = tmpclk;
149 
150  pEMC->DYNAMICCONTROL = 0x00000083; /* Issue MODE command */
151 
152  for (ChipSelect = 0; ChipSelect < 4; ChipSelect++) {
153  /*uint32_t burst_length;*/
154  uint32_t DynAddr;
155  uint8_t Col_len;
156 
157  Col_len = getColsLen(Dynamic_Config->DevConfig[ChipSelect].DynConfig);
158  /* get bus wide: if 32bit, len is 4 else if 16bit len is 2 */
159  /* burst_length = 1 << ((((Dynamic_Config->DynConfig[ChipSelect] >> 14) & 1)^1) +1); */
160  if (Dynamic_Config->DevConfig[ChipSelect].DynConfig & (1 << EMC_DYN_CONFIG_DATA_BUS_WIDTH_BIT)) {
161  /*32bit bus */
162  /*burst_length = 2;*/
163  Col_len += 2;
164  }
165  else {
166  /*burst_length = 4;*/
167  Col_len += 1;
168  }
169  DynAddr = Dynamic_Config->DevConfig[ChipSelect].BaseAddr;
170 
171  if (DynAddr != 0) {
172  uint32_t temp;
173  uint32_t ModeRegister;
174  ModeRegister = Dynamic_Config->DevConfig[ChipSelect].ModeRegister;
175  temp = *((volatile uint32_t *) (DynAddr | (ModeRegister << Col_len)));
176  temp = temp;
177  }
178  }
179  pEMC->DYNAMICCONTROL = 0x00000000; /* Issue NORMAL command */
180 
181  /* enable buffers */
182  pEMC->DYNAMICCONFIG0 |= 1 << 19;
183  pEMC->DYNAMICCONFIG1 |= 1 << 19;
184  pEMC->DYNAMICCONFIG2 |= 1 << 19;
185  pEMC->DYNAMICCONFIG3 |= 1 << 19;
186 }

192 {

194  EMC_Reg_add->STATICCONFIG0 = Static_Config->Config;
195  EMC_Reg_add->STATICWAITWEN0 = convertTimmingParam(EMC_Clock, Static_Config->WaitWen, 1);
196  EMC_Reg_add->STATICWAITOEN0 = convertTimmingParam(EMC_Clock, Static_Config->WaitOen, 0);
197  EMC_Reg_add->STATICWAITRD0 = convertTimmingParam(EMC_Clock, Static_Config->WaitRd, 1);
198  EMC_Reg_add->STATICWAITPAG0 = convertTimmingParam(EMC_Clock, Static_Config->WaitPage, 1);
199  EMC_Reg_add->STATICWAITWR0 = convertTimmingParam(EMC_Clock, Static_Config->WaitWr, 2);
200  EMC_Reg_add->STATICWAITTURN0 = convertTimmingParam(EMC_Clock, Static_Config->WaitTurn, 1);
