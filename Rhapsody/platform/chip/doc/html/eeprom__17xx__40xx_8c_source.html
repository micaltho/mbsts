102 /* Erase data in page register */
103 void Chip_EEPROM_ErasePageRegister(LPC_EEPROM_T *pEEPROM)
104 {
105  uint32_t i = 0;
106 
107  Chip_EEPROM_ClearIntStatus(pEEPROM, EEPROM_INT_ENDOFRW);
108 
109  Chip_EEPROM_SetCmd(pEEPROM, EEPROM_CMD_32BITS_WRITE);

120 /* Write data to page register */
121 uint32_t Chip_EEPROM_WritePageRegister(LPC_EEPROM_T *pEEPROM, uint16_t pageOffset,
122  uint8_t *pData, uint8_t wsize, uint32_t byteNum)
123 {
124  uint32_t i = 0;
125  uint32_t mask = (1 << (8 * wsize)) - 1;
126 
127  Chip_EEPROM_ClearIntStatus(pEEPROM, EEPROM_INT_ENDOFRW);
128 
129  if (wsize == 1) {
130  Chip_EEPROM_SetCmd(pEEPROM, EEPROM_CMD_8BITS_WRITE);
131  }
132  else if (wsize == 2) {
133  Chip_EEPROM_SetCmd(pEEPROM, EEPROM_CMD_16BITS_WRITE);
134  }
135  else {
136  Chip_EEPROM_SetCmd(pEEPROM, EEPROM_CMD_32BITS_WRITE);
137  }
138 
139  Chip_EEPROM_SetAddr(pEEPROM, 0, pageOffset);
140 
141  for (i = 0; i < byteNum; i += wsize) {
142  Chip_EEPROM_WriteData(pEEPROM, (*(uint32_t *) (&pData[i])) & mask);
143  Chip_EEPROM_WaitForIntStatus(pEEPROM, EEPROM_INT_ENDOFRW);
144  }
145 
146  return i;
147 }

158 /* Read data from non-volatile memory */
159 uint32_t Chip_EEPROM_ReadPage(LPC_EEPROM_T *pEEPROM,
160  uint16_t pageOffset,
161  uint16_t pageAddr,
162  uint8_t *pData,
163  uint8_t rsize,
164  uint32_t byteNum)
165 {
166  uint32_t i;
167  uint32_t mask = (1 << (8 * rsize)) - 1;
168 
169  Chip_EEPROM_ClearIntStatus(pEEPROM, EEPROM_INT_ENDOFRW);
170  Chip_EEPROM_SetAddr(pEEPROM, pageAddr, pageOffset);
171 
172  if (rsize == 1) {
173  Chip_EEPROM_SetCmd(pEEPROM, EEPROM_CMD_8BITS_READ | EEPROM_CMD_RDPREFETCH);
174  }
175  else if (rsize == 2) {
176  Chip_EEPROM_SetCmd(pEEPROM, EEPROM_CMD_16BITS_READ | EEPROM_CMD_RDPREFETCH);
177  }
178  else {
179  Chip_EEPROM_SetCmd(pEEPROM, EEPROM_CMD_32BITS_READ | EEPROM_CMD_RDPREFETCH);
180  }
181 
182  /* read and store data in buffer */
183  for (i = 0; i < byteNum; i += rsize) {
184  (*(uint32_t *) (&pData[i]) ) &= ~mask;
185  (*(uint32_t *) (&pData[i]) ) |= (Chip_EEPROM_ReadData(pEEPROM) & mask);
186  Chip_EEPROM_WaitForIntStatus(pEEPROM, EEPROM_INT_ENDOFRW);
187  }
188  return i;
189 }

191 /* Write data to EEPROM at specific address */
192 Status Chip_EEPROM_Write(LPC_EEPROM_T *pEEPROM,
193  uint16_t pageOffset,
194  uint16_t pageAddr,
195  void *pData,
196  EEPROM_RWSIZE_T wsize,
197  uint32_t byteNum)
198 {
199  uint32_t wTotalByteNum = 0;
200  uint32_t wOffset = (pageOffset & (EEPROM_PAGE_SIZE - 1));
201  uint32_t wByteNum = EEPROM_PAGE_SIZE - wOffset;
202  while (byteNum) {
203  if (wByteNum > byteNum) {
204  wByteNum = byteNum;
205  }
206  /* update data to page register */
207  Chip_EEPROM_WritePageRegister(pEEPROM, wOffset,
208  &((uint8_t *) pData)[wTotalByteNum], (uint8_t) wsize, wByteNum);
209  Chip_EEPROM_EraseProgramPage(pEEPROM, pageAddr);
210  wTotalByteNum += wByteNum;
211  byteNum -= wByteNum;
212 
213  /* Change to next page */
214  pageAddr++;
215  wOffset = 0;
216  wByteNum = EEPROM_PAGE_SIZE;
217  }
218  return SUCCESS;
219 }
