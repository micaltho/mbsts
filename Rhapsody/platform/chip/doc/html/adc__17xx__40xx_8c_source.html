59 /* Get divider value */
60 STATIC uint8_t getClkDiv(LPC_ADC_T *pADC, bool burstMode, uint32_t adcRate, uint8_t clks)
61 {
62  uint32_t adcBlockFreq;
63  uint32_t fullAdcRate;
64  uint8_t div;
65 
66  /* The APB clock (PCLK_ADC0) is divided by (CLKDIV+1) to produce the clock for
67  A/D converter, which should be less than or equal to 4.5MHz.
68  A fully conversion requires (bits_accuracy+1) of these clocks.
69  ADC Clock = PCLK_ADC0 / (CLKDIV + 1);
70  ADC rate = ADC clock / (the number of clocks required for each conversion);
71  */
72 #if defined(CHIP_LPC175X_6X)
73  adcBlockFreq = Chip_Clock_GetPeripheralClockRate(SYSCTL_PCLK_ADC);
74 #else
75  adcBlockFreq = Chip_Clock_GetPeripheralClockRate();
76 #endif
77 #if defined(ADC_ACC_12BITS)
78  fullAdcRate = adcRate * getFullConvClk();
79 #else
80  if (burstMode) {
81  fullAdcRate = adcRate * clks;
82  }
83  else {
84  fullAdcRate = adcRate * getFullConvClk();
85  }
86 #endif
87  /* Get the round value by fomular: (2*A + B)/(2*B) */
88  div = ((adcBlockFreq * 2 + fullAdcRate) / (fullAdcRate * 2)) - 1;
89  return div;
90 }

118 /* Initialize the ADC peripheral and the ADC setup structure to default value */
119 void Chip_ADC_Init(LPC_ADC_T *pADC, ADC_CLOCK_SETUP_T *ADCSetup)
120 {
121  uint8_t div;
122  uint32_t cr = 0;
123  uint32_t clk;
124 
125  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_ADC);
126 
127 #if defined(ADC_TRIM_SUPPORT)
128  pADC->ADTRM = 0xF00;
129 #endif
130  pADC->INTEN = 0; /* Disable all interrupts */
131 
132  cr |= ADC_CR_PDN;
133 
134  ADCSetup->adcRate = ADC_MAX_SAMPLE_RATE;
135  ADCSetup->bitsAccuracy = 0; /* LPC17xx/40xx doesn't support this setting */
136  clk = 0;
137  ADCSetup->burstMode = false;
138  div = getClkDiv(pADC, false, ADCSetup->adcRate, clk);
139  cr |= ADC_CR_CLKDIV(div);
140 #if !defined(ADC_ACC_12BITS)
141  cr |= ADC_CR_BITACC(ADCSetup->bitsAccuracy);
142 #endif /*defined(ADC_ACC_12BITS)*/
143  pADC->CR = cr;
144 }

160 /* Get ADC Channel status from ADC data register */
161 FlagStatus Chip_ADC_ReadStatus(LPC_ADC_T *pADC, uint8_t channel, uint32_t StatusType)
162 {
163  switch (StatusType) {
164  case ADC_DR_DONE_STAT:
165  return (pADC->STAT & (1UL << channel)) ? SET : RESET;
166 
167  case ADC_DR_OVERRUN_STAT:
168  channel += 8;
169  return (pADC->STAT & (1UL << channel)) ? SET : RESET;
170 
171  case ADC_DR_ADINT_STAT:
172  return pADC->STAT >> 16 ? SET : RESET;
173 
174  default:
175  break;
176  }
177  return RESET;
178 }

206 void Chip_ADC_SetSampleRate(LPC_ADC_T *pADC, ADC_CLOCK_SETUP_T *ADCSetup, uint32_t rate)
207 {
