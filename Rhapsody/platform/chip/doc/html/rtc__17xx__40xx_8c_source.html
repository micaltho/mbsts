50 /* Initialize the RTC peripheral */
51 void Chip_RTC_Init(LPC_RTC_T *pRTC)
52 {
53  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_RTC);
54 
55  /* Disable RTC */
56  Chip_RTC_Enable(pRTC, DISABLE);
57 
58  /* Disable Calibration */
59  Chip_RTC_CalibCounterCmd(pRTC, DISABLE);
60 
61  /* Reset RTC Clock */
62  Chip_RTC_ResetClockTickCounter(pRTC);
63 
64  /* Clear counter increment and alarm interrupt */
65  pRTC->ILR = RTC_IRL_RTCCIF | RTC_IRL_RTCALF;
66  while (pRTC->ILR != 0) {}
67 
68  /* Clear all register to be default */
69  pRTC->CIIR = 0x00;
70  pRTC->AMR = 0xFF;
71  pRTC->CALIBRATION = 0x00;
72 }

81 /* Reset clock tick counter in the RTC peripheral */
82 void Chip_RTC_ResetClockTickCounter(LPC_RTC_T *pRTC)
83 {
84  do {
85  /* Reset RTC clock*/
86  pRTC->CCR |= RTC_CCR_CTCRST;
87  } while ((pRTC->CCR & RTC_CCR_CTCRST) != RTC_CCR_CTCRST);
88 
89  do {
90  /* Finish resetting RTC clock */
91  pRTC->CCR &= (~RTC_CCR_CTCRST) & RTC_CCR_BITMASK;
92  } while (pRTC->CCR & RTC_CCR_CTCRST);
93 }

97 {
98  if (NewState == ENABLE) {

100  pRTC->CCR |= RTC_CCR_CLKEN;

102  }
103  else {
104  pRTC->CCR &= (~RTC_CCR_CLKEN) & RTC_CCR_BITMASK;
105  }

133 /* Set full time in the RTC peripheral */
134 void Chip_RTC_SetFullTime(LPC_RTC_T *pRTC, RTC_TIME_T *pFullTime)
135 {
136  RTC_TIMEINDEX_T i;
137  uint32_t ccr_val = pRTC->CCR;
138 
139  /* Temporarily disable */
140  if (ccr_val & RTC_CCR_CLKEN) {
141  pRTC->CCR = ccr_val & (~RTC_CCR_CLKEN) & RTC_CCR_BITMASK;
142  }
143 
144  /* Date time setting */
145  for (i = RTC_TIMETYPE_SECOND; i < RTC_TIMETYPE_LAST; i++) {
146  pRTC->TIME[i] = pFullTime->time[i];
147  }
148 
149  /* Restore to old setting */
150  pRTC->CCR = ccr_val;
151 }

153 /* Get full time from the RTC peripheral */
154 void Chip_RTC_GetFullTime(LPC_RTC_T *pRTC, RTC_TIME_T *pFullTime)
155 {
156  RTC_TIMEINDEX_T i;
157  uint32_t secs = 0xFF;
158 
159  /* Read full time, but verify second tick didn't change during the read. If
160  it did, re-read the time again so it will be consistent across all fields. */
161  while (secs != pRTC->TIME[RTC_TIMETYPE_SECOND]) {
162  secs = pFullTime->time[RTC_TIMETYPE_SECOND] = pRTC->TIME[RTC_TIMETYPE_SECOND];
163  for (i = RTC_TIMETYPE_MINUTE; i < RTC_TIMETYPE_LAST; i++) {
164  pFullTime->time[i] = pRTC->TIME[i];
165  }
166  }
167 }
