70 /* Basic Ethernet interface initialization */
71 void Chip_ENET_Init(LPC_ENET_T *pENET, bool useRMII)
72 {
73  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_ENET);
74  resetENET(pENET);
75 
76  /* Initial MAC configuration for full duplex,
77  100Mbps, inter-frame gap use default values */
78  pENET->MAC.MAC1 = ENET_MAC1_PARF;
79  pENET->MAC.MAC2 = ENET_MAC2_FULLDUPLEX | ENET_MAC2_CRCEN | ENET_MAC2_PADCRCEN;
80 
81  if (useRMII) {
82  pENET->CONTROL.COMMAND = ENET_COMMAND_FULLDUPLEX | ENET_COMMAND_PASSRUNTFRAME | ENET_COMMAND_RMII;
83  }
84  else {
85  pENET->CONTROL.COMMAND = ENET_COMMAND_FULLDUPLEX | ENET_COMMAND_PASSRUNTFRAME;
86  }
87 
88  pENET->MAC.IPGT = ENET_IPGT_FULLDUPLEX;
89  pENET->MAC.IPGR = ENET_IPGR_P2_DEF;
90  pENET->MAC.SUPP = ENET_SUPP_100Mbps_SPEED;
91  pENET->MAC.MAXF = ENET_ETH_MAX_FLEN;
92  pENET->MAC.CLRT = ENET_CLRT_DEF;
93 
94  /* Setup default filter */
95  pENET->CONTROL.COMMAND |= ENET_COMMAND_PASSRXFILTER;
96 
97  /* Clear all MAC interrupts */
98  pENET->MODULE_CONTROL.INTCLEAR = 0xFFFF;
99 
100  /* Disable MAC interrupts */
101  pENET->MODULE_CONTROL.INTENABLE = 0;
102 }

121 void Chip_ENET_SetupMII(LPC_ENET_T *pENET, uint32_t div, uint8_t addr)
122 {

133 /* Find the divider index for a desired MII clock rate */
134 uint32_t Chip_ENET_FindMIIDiv(LPC_ENET_T *pENET, uint32_t clockRate)
135 {
136  uint32_t tmp, divIdx = 0;
137 
138  /* Find desired divider value */
139  tmp = Chip_Clock_GetENETClockRate() / clockRate;
140 
141  /* Determine divider index from desired divider */
142  for (divIdx = 0; divIdx < (sizeof(EnetClkDiv) / sizeof(EnetClkDiv[0])); divIdx++) {
143  /* Closest index, but not higher than desired rate */
144  if (EnetClkDiv[divIdx] >= tmp) {
145  return divIdx;
146  }
147  }
148 
149  /* Use maximum divider index */
150  return (sizeof(EnetClkDiv) / sizeof(EnetClkDiv[0])) - 1;
151 }

219 /* Get status for the descriptor list */
220 ENET_BUFF_STATUS_T Chip_ENET_GetBufferStatus(LPC_ENET_T *pENET,
221  uint16_t produceIndex,
222  uint16_t consumeIndex,
223  uint16_t buffSize)
224 {
225  /* Empty descriptor list */
226  if (consumeIndex == produceIndex) {
227  return ENET_BUFF_EMPTY;
228  }
229 
230  /* Full descriptor list */
231  if ((consumeIndex == 0) &&
232  (produceIndex == (buffSize - 1))) {
233  return ENET_BUFF_FULL;
234  }
235 
236  /* Wrap-around, full descriptor list */
237  if (consumeIndex == (produceIndex + 1)) {
238  return ENET_BUFF_FULL;
239  }
240 
241  return ENET_BUFF_PARTIAL_FULL;
242 }

244 /* Get the number of descriptor filled */
245 uint32_t Chip_ENET_GetFillDescNum(LPC_ENET_T *pENET, uint16_t produceIndex, uint16_t consumeIndex, uint16_t buffSize)
246 {
247  /* Empty descriptor list */
248  if (consumeIndex == produceIndex) {
249  return 0;
250  }
251 
252  if (consumeIndex > produceIndex) {
253  return (buffSize - consumeIndex) + produceIndex;
254  }
255 
256  return produceIndex - consumeIndex;
257 }
